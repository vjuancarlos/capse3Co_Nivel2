218 char* itoa(int value, char* result, int base) {
219  // check that the base if valid
220  if (base < 2 || base > 36) { *result = '\0'; return result; }
221 
222  char* ptr = result, *ptr1 = result, tmp_char;
223  int tmp_value;
224 
225  do {
226  tmp_value = value;
227  value /= base;
228  *ptr++ = "zyxwvutsrqponmlkjihgfedcba9876543210123456789abcdefghijklmnopqrstuvwxyz" [35 + (tmp_value - value * base)];
229  } while ( value );
230 
231  // Apply negative sign
232  if (tmp_value < 0) *ptr++ = '-';
233  *ptr-- = '\0';
234  while(ptr1 < ptr) {
235  tmp_char = *ptr;
236  *ptr--= *ptr1;
237  *ptr1++ = tmp_char;
238  }
239  return result;
240 }
